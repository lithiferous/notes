wcss = []
for i in range(1,25):
    kmeans = KMeans(n_clusters=i,init='k-means++',max_iter=300,n_init=24,random_state=0, n_jobs=8)
    kmeans.fit(df)
    wcss.append(kmeans.inertia_)
    
plt.plot(range(1,25),wcss)
plt.title('The Elbow Method')
plt.xlabel('Number of clusters')
plt.ylabel('WCSS')
plt.savefig('1elbow.png')
plt.show()
